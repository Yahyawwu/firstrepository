There are also a range of other NYT APIs which you can find here if these are of greater interest to you. def obtain_monthly_counts(search_term, key, start_year, end_year):
    
    params = {"q": search_term,
              "api-key": key}
    
    all_counts = []
    
    for year in range(start_year, end_year+1):
        
            for month in range(1,13):
                
                if month<10:
                    month = "0"+str(month)

                params["begin_date"] = f"{year}{month}01"
                params["end"] = f"{year}{month}31"

                r = requests.get("https://api.nytimes.com/svc/search/v2/articlesearch.json", params=params)
                data = r.json()
                count = data["response"]["meta"]["hits"]

                # Wait 7 seconds between requests to hace some buffer as only 10 requests
                # per minute are allowed
                time.sleep(7)

                all_counts.append(count)
                
    counts_series = pd.Series(all_counts)
    counts_series.index = pd.date_range(pd.to_datetime(f"{start_year}0101"), periods=(end_year-start_year+1)*12, freq='M')
            
    return counts_series
